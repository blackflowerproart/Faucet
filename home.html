async function init() { // 1. التحقق من وجود جلسة نشطة const { data: { session } } = await _supabase.auth.getSession(); if (!session) { window.location.replace("index.html"); return; } const user = session.user; let displayName = "USER"; try { // 2. محاولة جلب الاسم من جدول قاعدة البيانات (للتسجيل اليدوي) const { data: profile } = await _supabase .from('blackflowerproart') .select('username, balance') .eq('user_id', user.id) .single(); if (profile && profile.username) { displayName = profile.username; document.getElementById('bfp-val').innerText = profile.balance.toLocaleString(); } else { // 3. إذا لم يوجد بروفايل (دخول اجتماعي)، نجلب الاسم من MetaData // جرب جلب الاسم الكامل، أو اسم المستخدم، أو حتى أول جزء من الإيميل displayName = user.user_metadata.full_name || user.user_metadata.name || user.user_metadata.custom_claims?.global_name || // خاص لديسكورد user.email.split('@')[0]; document.getElementById('bfp-val').innerText = "0"; // رصيد افتراضي جديد } // عرض الاسم في الواجهة document.getElementById('user-name').innerText = displayName.toUpperCase(); } catch (err) { console.error("Error fetching user data:", err); // في حالة الخطأ، اعرض الإيميل كاسم احتياطي document.getElementById('user-name').innerText = user.email.split('@')[0].toUpperCase(); } // تحميل الإعلانات المتاحة await loadAds(user.id); // إظهار الصفحة بعد انتهاء التحميل document.body.style.visibility = "visible"; }
