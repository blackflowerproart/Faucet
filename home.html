// داخل سكريبت صفحة home.html async function fetchUserProfile() { const { data: { user } } = await _supabase.auth.getUser(); // جلب البيانات حصراً للمستخدم المسجل حالياً باستخدام id الخاص به const { data, error } = await _supabase .from('profiles') .select('username, email, wallets(balance)') .eq('id', user.id) // التأكد من جلب بيانات المستخدم الحالي فقط .single(); if (data) { document.getElementById('display-name').innerText = data.username; document.getElementById('display-email').innerText = data.email; // عرض الرصيد الخاص بالمستخدم من جدول المحفظة المرتبط document.getElementById('balance').innerText = data.wallets ? data.wallets.balance.toFixed(2) : "0.00"; } }
