// أضف هذا المنطق داخل دالة الإرسال في صفحة AdControlManeger async function handleAddAd() { const title = document.getElementById('adTitle').value; const url = document.getElementById('adUrl').value; const durationType = document.getElementById('durationType').value; // 10s or 20s const days = parseInt(document.getElementById('adDays').value); // 1, 7, 30 days if (days < 1) { alert("أقل مدة هي يوم واحد (24 ساعة)"); return; } // حساب السعر: 20 ثانية بـ 100 يومياً، 10 ثواني بـ 50 يومياً const dailyPrice = (durationType === "20") ? 100 : 50; const totalPrice = dailyPrice * days; // حساب تاريخ الانتهاء const expiryDate = new Date(); expiryDate.setHours(expiryDate.getHours() + (days * 24)); const { data, error } = await _supabase.from('ads').insert([{ title: title, url: url, duration: parseInt(durationType), expiry_at: expiryDate.toISOString(), // تاريخ الانتهاء daily_price: dailyPrice, total_paid: totalPrice, created_at: new Date().toISOString() }]); if (!error) alert(`تم إضافة الإعلان بنجاح. التكلفة الإجمالية: ${totalPrice} BFP`); }
